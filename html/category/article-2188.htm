import tensorflow as tf  input_x = tf.Variable([1, 2, 3], dtype=tf.float32) ones = tf.ones([3,], dtype=tf.float32) concat = tf.concat([input_x, ones], axis=0) sess = tf.Session() sess.run(tf.global_variables_initializer()) sess.run(concat)   #output: array([1., 2., 3., 1., 1., 1.], dtype=float32)
